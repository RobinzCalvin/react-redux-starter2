import { combineReducers, Reducer } from 'redux';
import { makeCommunicationReducer } from 'shared/helpers/redux';

import initialState from '../initial';
import { IReduxState } from '../../namespace';

function dataReducer(state: IReduxState['data'] = initialState.data, action /* this will be type of actions */):
  IReduxState['data'] {
  switch (action.type) {
    default: {
      return state;
    }
  }
}

function uiReducer(state: IReduxState['ui'] = initialState.ui, action: any): IReduxState['ui'] {
  switch (action.type) {
    default: {
      return state;
    }
  }
}

const reducer: Reducer<IReduxState> = combineReducers<IReduxState>({
  communication: combineReducers<IReduxState['communication']>({}),
  data: dataReducer,
  ui: uiReducer,
});

export default reducer;
